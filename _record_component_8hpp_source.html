117 inline std::shared_ptr< T >
118 RecordComponent::loadChunk(Offset const& o, Extent const& e, double targetUnitSI)
119 {
120  uint64_t numPoints = 1u;
121  for( auto const& dimensionSize : e )
122  numPoints *= dimensionSize;
123 
124  auto newData = std::shared_ptr<T>(new T[numPoints], []( T *p ){ delete [] p; });
125  loadChunk(o, e, newData, targetUnitSI);

129 template< typename T >
130 inline void
131 RecordComponent::loadChunk(Offset const& o, Extent const& e, std::shared_ptr< T > data, double targetUnitSI)
132 {
133  if( !std::isnan(targetUnitSI) )
134  throw std::runtime_error("unitSI scaling during chunk loading not yet implemented");
135  Datatype dtype = determineDatatype(data);
136  if( dtype != getDatatype() )
137  if( !isSameInteger< T >( dtype ) && !isSameFloatingPoint< T >( dtype ) )
138  throw std::runtime_error("Type conversion during chunk loading not yet implemented");
139 
140  uint8_t dim = getDimensionality();
141  if( e.size() != dim || o.size() != dim )
142  throw std::runtime_error("Dimensionality of chunk and dataset do not match.");
143  Extent dse = getExtent();
144  for( uint8_t i = 0; i < dim; ++i )
145  if( dse[i] < o[i] + e[i] )
146  throw std::runtime_error("Chunk does not reside inside dataset (Dimension on index " + std::to_string(i)
147  + " - DS: " + std::to_string(dse[i])
148  + " - Chunk: " + std::to_string(o[i] + e[i])
149  + ")");
150  if( !data )
151  throw std::runtime_error("Unallocated pointer passed during chunk loading.");
152 
153  if( *m_isConstant )
154  {
155  uint64_t numPoints = 1u;
156  for( auto const& dimensionSize : e )
157  numPoints *= dimensionSize;
158 
159  T value = m_constantValue->get< T >();
160 
161  T* raw_ptr = data.get();
162  std::fill(raw_ptr, raw_ptr + numPoints, value);
163  } else
164  {
165  Parameter< Operation::READ_DATASET > dRead;
166  dRead.offset = o;
167  dRead.extent = e;
168  dRead.dtype = getDatatype();
169  dRead.data = std::static_pointer_cast< void >(data);
170  m_chunks->push(IOTask(this, dRead));
171  }
172 }

174 template< typename T >
175 inline void
176 RecordComponent::storeChunk(Offset o, Extent e, std::shared_ptr<T> data)
177 {
178  if( *m_isConstant )
179  throw std::runtime_error("Chunks can not be written for a constant RecordComponent.");
180  if( !data )
181  throw std::runtime_error("Unallocated pointer passed during chunk store.");
182  Datatype dtype = determineDatatype(data);
183  if( dtype != getDatatype() )
184  {
185  std::ostringstream oss;
186  oss << "Datatypes of chunk data ("
187  << dtype
188  << ") and dataset ("
189  << getDatatype()
190  << ") do not match.";
191  throw std::runtime_error(oss.str());
192  }
193  uint8_t dim = getDimensionality();
194  if( e.size() != dim || o.size() != dim )
195  throw std::runtime_error("Dimensionality of chunk and dataset do not match.");
196  Extent dse = getExtent();
197  for( uint8_t i = 0; i < dim; ++i )
198  if( dse[i] < o[i] + e[i] )
199  throw std::runtime_error("Chunk does not reside inside dataset (Dimension on index " + std::to_string(i)
200  + " - DS: " + std::to_string(dse[i])
201  + " - Chunk: " + std::to_string(o[i] + e[i])
202  + ")");
203 
204  Parameter< Operation::WRITE_DATASET > dWrite;
205  dWrite.offset = o;
206  dWrite.extent = e;
207  dWrite.dtype = dtype;
208  /* std::static_pointer_cast correctly reference-counts the pointer */
209  dWrite.data = std::static_pointer_cast< void const >(data);
210  m_chunks->push(IOTask(this, dWrite));
211 }
