206 template< typename T >
207 inline std::shared_ptr< T >
208 RecordComponent::loadChunk(Offset o, Extent e, double targetUnitSI)
209 {
210  uint8_t dim = getDimensionality();
211 
212  // default arguments
213  // offset = {0u}: expand to right dim {0u, 0u, ...}
214  Offset offset = o;
215  if( o.size() == 1u && o.at(0) == 0u && dim > 1u )
216  offset = Offset(dim, 0u);
217 
218  // extent = {-1u}: take full size
219  Extent extent(dim, 1u);
220  if( e.size() == 1u && e.at(0) == -1u )
221  {
222  extent = getExtent();
223  for( uint8_t i = 0u; i < dim; ++i )
224  extent[i] -= offset[i];
225  }
226  else
227  extent = e;
228 
229  uint64_t numPoints = 1u;
230  for( auto const& dimensionSize : extent )
231  numPoints *= dimensionSize;
232 
233  auto newData = std::shared_ptr<T>(new T[numPoints], []( T *p ){ delete [] p; });
234  loadChunk(newData, offset, extent, targetUnitSI);
235  return newData;
236 }

238 template< typename T >
239 inline void
240 RecordComponent::loadChunk(std::shared_ptr< T > data, Offset o, Extent e, double targetUnitSI)
241 {
242  if( !std::isnan(targetUnitSI) )
243  throw std::runtime_error("unitSI scaling during chunk loading not yet implemented");
244  Datatype dtype = determineDatatype(data);
245  if( dtype != getDatatype() )
246  if( !isSameInteger< T >( dtype ) && !isSameFloatingPoint< T >( dtype ) )
247  throw std::runtime_error("Type conversion during chunk loading not yet implemented");
248 
249  uint8_t dim = getDimensionality();
250 
251  // default arguments
252  // offset = {0u}: expand to right dim {0u, 0u, ...}
253  Offset offset = o;
254  if( o.size() == 1u && o.at(0) == 0u && dim > 1u )
255  offset = Offset(dim, 0u);
256 
257  // extent = {-1u}: take full size
258  Extent extent(dim, 1u);
259  if( e.size() == 1u && e.at(0) == -1u )
260  {
261  extent = getExtent();
262  for( uint8_t i = 0u; i < dim; ++i )
263  extent[i] -= offset[i];
264  }
265  else
266  extent = e;
267 
268  if( extent.size() != dim || offset.size() != dim )
269  {
270  std::ostringstream oss;
271  oss << "Dimensionality of chunk ("
272  << "offset=" << offset.size() << "D, "
273  << "extent=" << extent.size() << "D) "
274  << "and record component ("
275  << int(dim) << "D) "
276  << "do not match.";
277  throw std::runtime_error(oss.str());
278  }
279  Extent dse = getExtent();
280  for( uint8_t i = 0; i < dim; ++i )
281  if( dse[i] < offset[i] + extent[i] )
282  throw std::runtime_error("Chunk does not reside inside dataset (Dimension on index " + std::to_string(i)
283  + " - DS: " + std::to_string(dse[i])
284  + " - Chunk: " + std::to_string(offset[i] + extent[i])
285  + ")");
286  if( !data )
287  throw std::runtime_error("Unallocated pointer passed during chunk loading.");
288 
289  if( *m_isConstant )
290  {
291  uint64_t numPoints = 1u;
292  for( auto const& dimensionSize : extent )
293  numPoints *= dimensionSize;
294 
295  T value = m_constantValue->get< T >();
296 
297  T* raw_ptr = data.get();
298  std::fill(raw_ptr, raw_ptr + numPoints, value);
299  } else
300  {
301  Parameter< Operation::READ_DATASET > dRead;
302  dRead.offset = offset;
303  dRead.extent = extent;
304  dRead.dtype = getDatatype();
305  dRead.data = std::static_pointer_cast< void >(data);
306  m_chunks->push(IOTask(this, dRead));
307  }
308 }

310 template< typename T >
311 inline void
312 RecordComponent::storeChunk(std::shared_ptr<T> data, Offset o, Extent e)
313 {
314  if( *m_isConstant )
315  throw std::runtime_error("Chunks cannot be written for a constant RecordComponent.");
316  if( *m_isEmpty )
317  throw std::runtime_error("Chunks cannot be written for an empty RecordComponent.");
318  if( !data )
319 throw std::runtime_error("Unallocated pointer passed during chunk store.");
320  Datatype dtype = determineDatatype(data);
321  if( dtype != getDatatype() )
322  {
323  std::ostringstream oss;
324  oss << "Datatypes of chunk data ("
325  << dtype
326  << ") and record component ("
327  << getDatatype()
328  << ") do not match.";
329  throw std::runtime_error(oss.str());
330  }
331  uint8_t dim = getDimensionality();
332  if( e.size() != dim || o.size() != dim )
333  {
334  std::ostringstream oss;
335  oss << "Dimensionality of chunk ("
336  << "offset=" << o.size() << "D, "
337  << "extent=" << e.size() << "D) "
338  << "and record component ("
339  << int(dim) << "D) "
340  << "do not match.";
341  throw std::runtime_error(oss.str());
342  }
343  Extent dse = getExtent();
344  for( uint8_t i = 0; i < dim; ++i )
345  if( dse[i] < o[i] + e[i] )
346  throw std::runtime_error("Chunk does not reside inside dataset (Dimension on index " + std::to_string(i)
347  + " - DS: " + std::to_string(dse[i])
348  + " - Chunk: " + std::to_string(o[i] + e[i])
349  + ")");
350 
351  Parameter< Operation::WRITE_DATASET > dWrite;
352  dWrite.offset = o;
353  dWrite.extent = e;
354  dWrite.dtype = dtype;
355  /* std::static_pointer_cast correctly reference-counts the pointer */
356  dWrite.data = std::static_pointer_cast< void const >(data);
357  m_chunks->push(IOTask(this, dWrite));
358 }
