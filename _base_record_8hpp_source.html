97 template< typename T_elem >
98 inline typename BaseRecord< T_elem >::mapped_type&
99 BaseRecord< T_elem >::operator[](key_type const& key)
100 {
101  auto it = this->find(key);
102  if( it != this->end() )
103  return it->second;
104  else
105  {
106  bool scalar = (key == RecordComponent::SCALAR);
107  if( (scalar && !Container< T_elem >::empty() && !*m_containsScalar) || (*m_containsScalar && !scalar) )
108  throw std::runtime_error("A scalar component can not be contained at "
109  "the same time as one or more regular components.");
110 
111  mapped_type& ret = Container< T_elem >::operator[](key);
112  if( scalar )
113  {
114  *m_containsScalar = true;
115  ret.m_writable->parent = this->m_writable->parent;
116  ret.parent = this->parent;
117  }
118  return ret;
119  }
120 }

147 template< typename T_elem >
148 inline typename BaseRecord< T_elem >::size_type
149 BaseRecord< T_elem >::erase(key_type const& key)
150 {
151  bool scalar = (key == RecordComponent::SCALAR);
152  size_type res;
153  if( !scalar || (scalar && *this->at(key).m_isConstant) )
154  res = Container< T_elem >::erase(key);
155  else
156  {
157  mapped_type& rc = this->find(RecordComponent::SCALAR)->second;
158  if( rc.written )
159  {
160  Parameter< Operation::DELETE_DATASET > dDelete;
161  dDelete.name = ".";
162  this->IOHandler->enqueue(IOTask(&rc, dDelete));
163  this->IOHandler->flush();
164  }
165  res = Container< T_elem >::erase(key);
166  }
167 
168  if( scalar )
169  {
170  this->written = false;
171  this->m_writable->abstractFilePosition.reset();
172  *this->m_containsScalar = false;
173  }
174  return res;
175 }

184 template< typename T_elem >
185 inline void
186 BaseRecord< T_elem >::readBase()
187 {
188  using DT = Datatype;
189  Parameter< Operation::READ_ATT > aRead;
190 
191  aRead.name = "unitDimension";
192  this->IOHandler->enqueue(IOTask(this, aRead));
193  this->IOHandler->flush();
194  if( *aRead.dtype == DT::ARR_DBL_7 )
195  this->setAttribute("unitDimension", Attribute(*aRead.resource).template get< std::array< double, 7 > >());
196  else if( *aRead.dtype == DT::VEC_DOUBLE )
197  {
198  auto vec = Attribute(*aRead.resource).template get< std::vector< double > >();
199  if( vec.size() == 7 )
200  {
201  std::array< double, 7 > arr;
202  std::copy(vec.begin(),
203  vec.end(),
204  arr.begin());
205  this->setAttribute("unitDimension", arr);
206  } else
207  throw std::runtime_error("Unexpected Attribute datatype for 'unitDimension'");
208  }
209  else
210  throw std::runtime_error("Unexpected Attribute datatype for 'unitDimension'");
211 
212  aRead.name = "timeOffset";
213  this->IOHandler->enqueue(IOTask(this, aRead));
214  this->IOHandler->flush();
215  if( *aRead.dtype == DT::FLOAT )
216  this->setAttribute("timeOffset", Attribute(*aRead.resource).template get< float >());
217  else if( *aRead.dtype == DT::DOUBLE )
218  this->setAttribute("timeOffset", Attribute(*aRead.resource).template get< double >());
219  else
220  throw std::runtime_error("Unexpected Attribute datatype for 'timeOffset'");
221 }
